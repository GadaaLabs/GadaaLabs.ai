## Part 1: Classes from Scratch

### The Class Statement and `__init__`

A class is a blueprint for objects. When you call `MyClass()`, Python allocates a new instance (`__new__`), then calls `__init__` to initialize it. `self` is not a keyword — it is simply the first parameter of every instance method, and Python automatically passes the instance as that first argument.

<PythonPlayground code={String.raw`class Point:

    def __init__(self, x, y):
        # Instance attributes live in the instance's __dict__

        self.y = y

    def distance_to(self, other):
        dx = self.x - other.x
        dy = self.y - other.y
        return (dx**2 + dy**2) ** 0.5

### Comparison Operators and `@total_ordering`

<PythonPlayground code={String.raw`from functools import total_ordering

@total_ordering   # only need __eq__ and ONE comparison method
class Version:
    """Semantic version comparison."""
    def __init__(self, version_str):
        parts = version_str.split(".")
        self.major = int(parts[0])
        self.minor = int(parts[1]) if len(parts) > 1 else 0
        self.patch = int(parts[2]) if len(parts) > 2 else 0

    def _tuple(self):
        return (self.major, self.minor, self.patch)

    def __eq__(self, other):
        if not isinstance(other, Version):
            return NotImplemented
        return self._tuple() == other._tuple()

    def __lt__(self, other):
        if not isinstance(other, Version):
            return NotImplemented
        return self._tuple() < other._tuple()

    def __repr__(self):
        return f"Version('{self.major}.{self.minor}.{self.patch}')"

    def __hash__(self):
        return hash(self._tuple())   # required when __eq__ is defined

v1 = Version("1.0.0")
v2 = Version("1.2.3")
v3 = Version("2.0.0")
v4 = Version("1.2.3")

print(f"v2 == v4: {v2 == v4}")   # True
print(f"v3 > v2: {v3 > v2}")     # True — generated by @total_ordering
print(f"v1 <= v2: {v1 <= v2}")   # True — generated

versions = [Version("1.5.0"), Version("2.0.1"), Version("1.0.0"), Version("1.5.1")]
print(sorted(versions))

# Works in a set because __hash__ is defined

### Arithmetic Operators and Reflected Methods

<PythonPlayground code={String.raw`class Vector:
    def __init__(self, *components):
        self.components = tuple(components)

    def __repr__(self):
        return f"Vector{self.components}"

    def __len__(self):
        return len(self.components)

    def __add__(self, other):
        if isinstance(other, Vector):
            if len(self) != len(other):
                raise ValueError("Vectors must have same dimension")
            return Vector(*(a + b for a, b in zip(self.components, other.components)))
        return NotImplemented

    def __radd__(self, other):
        """Called when left operand does not support __add__ with Vector.
        Makes sum([v1, v2, v3]) work (sum starts with integer 0)."""
        if other == 0:
            return self
        return NotImplemented

    def __mul__(self, scalar):
        if isinstance(scalar, (int, float)):
            return Vector(*(c * scalar for c in self.components))
        return NotImplemented

    def __rmul__(self, scalar):
        return self.__mul__(scalar)   # 3 * v == v * 3

    def __neg__(self):
        return Vector(*(-c for c in self.components))

    def __abs__(self):
        return sum(c**2 for c in self.components) ** 0.5

    def __iadd__(self, other):
        if isinstance(other, Vector):
            self.components = tuple(a + b for a, b in zip(self.components, other.components))
            return self
        return NotImplemented

v2 = Vector(4, 5, 6)

print(-v1)
print(f"magnitude: {abs(v1):.4f}")

                self.add(item)

        return iter(self._data)

        return reversed(self._data)

### Method Resolution Order (MRO) — C3 Linearization

Python uses the **C3 linearization algorithm** to compute the MRO. The invariant: a class always appears before its parents, and if two classes share an ancestor, the one listed first in the inheritance list takes precedence. This guarantees the ancestor is called exactly once in the diamond problem.

<PythonPlayground code={String.raw`class A:
    def method(self):
        print("A.method")

class B(A):
    def method(self):
        print("B.method")
        super().method()   # super() follows MRO — NOT necessarily A

class C(A):
    def method(self):
        print("C.method")
        super().method()

class D(B, C):
    def method(self):
        print("D.method")
        super().method()

d = D()
d.method()
# Output order: D -> B -> C -> A (A called once)

print("\nMRO for D:")
for cls in D.__mro__:
    print(f"  {cls.__name__}")

# Each super() call advances to the NEXT class in the ORIGINAL instance's MRO.
# This is cooperative multiple inheritance.

# Mixin pattern — add functionality without deep inheritance
class JSONMixin:
    """Adds to_json() to any class with __dict__."""
    def to_json(self):
        import json
        return json.dumps(vars(self), default=str)

class LogMixin:
    """Adds log() to any class."""
    def log(self, msg):
        print(f"[{self.__class__.__name__}] {msg}")

class User(JSONMixin, LogMixin):
    def __init__(self, name, email):
        self.name = name
        self.email = email

u = User("Alice", "alice@example.com")
print(u.to_json())
u.log("User created")
print([cls.__name__ for cls in User.__mro__])

### Descriptors — The Engine Behind Properties

Any class defining `__get__`, `__set__`, or `__delete__` is a **descriptor**. Properties, classmethods, and staticmethods are all descriptors under the hood. Writing your own descriptor lets you reuse validation logic across multiple classes.

<PythonPlayground code={String.raw`class Validated:
    """Reusable descriptor for numeric range validation."""

    def __init__(self, min_val, max_val, doc=""):
        self.min_val = min_val
        self.max_val = max_val
        self.__doc__ = doc
        self.name = None

    def __set_name__(self, owner, name):
        # Called at class creation time — gives us the attribute name
        self.name = name

    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        return obj.__dict__.get(self.name, self.min_val)

    def __set__(self, obj, value):
        if not isinstance(value, (int, float)):
            raise TypeError(f"{self.name} must be a number")
        if not (self.min_val <= value <= self.max_val):
            raise ValueError(
                f"{self.name} must be between {self.min_val} and {self.max_val}, got {value}"
            )
        obj.__dict__[self.name] = value

    def __delete__(self, obj):
        obj.__dict__.pop(self.name, None)

class Sensor:
    temperature = Validated(-40, 125, "Temperature in Celsius")

        self.temperature = temp
        self.humidity = humidity
        self.pressure = pressure

        return f"Sensor(T={self.temperature}C, H={self.humidity}%, P={self.pressure}hPa)"

print(s)

s.temperature = 35

    s.temperature = 200
except ValueError as e:
    print(f"ValueError: {e}")
